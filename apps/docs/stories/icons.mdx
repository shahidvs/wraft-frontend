import { Meta, Title } from '@storybook/blocks'
import * as Icons from '@wraft/icon'
import React, { useState } from 'react';
import { SearchIcon } from "@wraft/icon";




<Meta title='Icons' />


<Title>Icons</Title>

<br />

## Overview

Wraft icons are simple, single-color glyphs that identify labels and actions across Audius. They can be paired with text or used alone. They are vector based and should consistent on a single vector layer framed in a 24x24 container. They have no background shape or fill.

<br />
{(() => {
  const [searchTerm, setSearchTerm] = useState('');
  
  const filteredIcons = Object.keys(Icons).filter(key =>
    key.toLowerCase().includes(searchTerm.toLowerCase())
  );

  return (
    <>
      <div style={{ 
        marginBottom: '20px',
        position: 'relative',
        maxWidth: '300px'
      }}>
        <SearchIcon 
          style={{
            position: 'absolute',
            left: '12px',
            top: '50%',
            transform: 'translateY(-50%)',
            color: '#666'
          }}
        />
        <input
          type="text"
          placeholder="Search Icons"
          value={searchTerm}
          onChange={(e) => setSearchTerm(e.target.value)}
          style={{
            padding: '8px 12px 8px 40px',
            border: '1px solid #ccc',
            borderRadius: '4px',
            width: '100%'
          }}
        />
      </div>

      <div style={{display: 'flex', gap:'18px', flexWrap: 'wrap'}}>
        {filteredIcons.map(key => {
          const Icon = Icons[key]
          return (
            <div key={key} style={{display: 'flex', flexDirection: 'column', alignItems: 'center'}}>
              <Icon />
              <div style={{paddingTop: '6px', paddingBottom: '6px'}}>{key}</div>
            </div>
          )
        })}
      </div>
    </>
  );
})()}
